\begin{empheq}[box=\tcbhighmath]{equation*}
  \begin{split}

    -\diffty \ddzz{(Cy_i)} +  \ddzuv{u}{c_i} + \ddt{c_i} + S_i = 0 \\
    -\diffty \ddz{(\ddzuv{C}{y_i})} + \ddzuv{u}{c_i}  + \ddt{c_i} + S_i = 0 \\
    -\diffty \left \{ \ddzudv{y_i}{C} + \ddzudv{C}{y_i} \right \}

  \end{split}
\end{empheq}

\begin{empheq}[box=\tcbhighmath]{equation*}
  \begin{split}
    -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
        + \ddzuv{u}{c_i}  + \ddt{c_i} + S_i = 0 \\
    -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
        + C y_i \ddz{u} + u \left \{ \ddzuv{C}{y_i} \right \} 
	+ C \ddt{y_i}  + y_i \ddt{C} + S_i = 0  \\
     C \left \{ \ddt{y_i} + y_i \ddz{u} + u \ddz{y_i} \right \}
	+ uy_i\ddz{C} + y_i\ddt{C}
        -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
        + S_i = 0 \\
     C \left \{ \ddt{y_i} + y_i \ddz{u} + u \ddz{y_i} \right \}
        + y_i \left\{ u\ddz{C} + \ddt{C} \right \}
        -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
        + S_i = 0 \\ 
  \end{split}
\end{empheq}



\begin{empheq}[box=\tcbhighmath]{equation*}
  \begin{split}
    \text{Rearranging}  \\
     C \left \{ \ddt{y_i} + u \ddz{y_i} \right \}
       + y_i \left\{ C\ddz{u} + u\ddz{C} + \ddt{C} \right \}
       -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
       + S_i = 0 \\ 
     C \left \{ \ddt{y_i} +
       u \ddz{y_i} \right \}
       + y_i \left \{ \ddz{(Cu)} + \ddt{C} \right \}
       -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
       + S_i = 0 
  \end{split}
\end{empheq}


\begin{empheq}[box=\tcbhighmath]{equation*}
  \begin{split}
    \text{Total Mass Blance} \\
      \ddt{C} + \ddz{(Cu)} + \sum{S_i} = 0 \\
      \text{Now Component Balance can be written as} \\
    C \left \{ \ddt{y_i} + u \ddz{y_i} \right \} - y_i \sum{S_i}
     -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
     + S_i = 0 \\
    C \left \{ \ddt{y_i} + u \ddz{y_i} \right \} 
        -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
      + S_i - y_i \sum{S_i}= 0 
    \end{split}
\end{empheq}


Another way
\begin{empheq}[box=\tcbhighmath]{equation*}
\begin{align}
    -\diffty \left \{ \ddzudv{y_i}{C} + \ddzudv{C}{y_i} \right \}  
    + \ddzuv{(Cu)}{y_i}  + \ddt{(Cy_i)} + S_i = 0  \\
    -\diffty \left \{ \ddzudv{y_i}{C} + \ddzudv{C}{y_i} \right \}  
    + y_i \ddz{(Cu)} + Cu \ddz{y_i}  + y_i \ddt{C} + C \ddt{y_i} + S_i = 0 
\end{align}
 \end{empheq}

\begin{empheq}[box=\tcbhighmath]{equation*}
  \begin{split}
      \text{Total Mass Blance }   \\
      \ddt{C} + \ddz{(Cu)} + \sum{S_i} = 0 \\
      \text{Component Balance} \\
    C \left \{ \ddt{y_i} + u \ddz{y_i} \right \} 
        -\diffty \left\{ C \ddzz{y_i} + 2 \ddz{C} \ddz{y_i} + y_i \ddzz{C} \right \}
      + S_i - y_i \sum{S_i}= 0 \\
      \text{where} \\
          S_i = \frac{1-\epsilon}{\epsilon} \sum_i \ddt{q_i} \\
          \ddt{q_i} = k_i (q_i^* - q_i)
    \end{split}
\end{empheq}


\begin{tcolorbox}[fonttitle=\sffamily\bfseries\large,

\begin{empheq}[box=\tcbhighmath]{equation*}
  \begin{split}
      \text{Total Mass Blance in dimensionless form}   \\
      \ddts{C^*} + \ddzs{(C^*u^*)} + \sum{S_i^*} = 0 \\
      \text{Component Balance} \\
    C^* \left \{ \ddts{y_i} + u^* \ddzs{y_i} \right \} 
        -\frac{1}{P_e} \left\{ C^* \ddzzs{y_i} + 2 \ddzs{C^*} \ddzs{y_i} + y_i \ddzzs{C^*} \right \}
      + S_i^* - y_i \sum{S_i}^*= 0 \\
      \text{where} \\
          S_i^* = \frac{1-\epsilon}{\epsilon} k_i\tau (q_i^{**} - q_i^*) \\
          C^* = C/C^{o} ;\  t^* = t/\tau \   z^* = z/L \  q_i^* = q_i/C^{o}; \ 
          P_e=\frac{u^oL}{\diffty} ; \  \tau = L/u^o
    \end{split}
\end{empheq}
\end{tcolorbox}
